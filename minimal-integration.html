📦 Integration Code

Here's how to add MotionScript to your website:

<!-- 1. Include Three.js and MotionScript --> <script src="https://cdnjs.cloudflare.com/ajax/libs/three.js/r128/three.min.js"></script> <script src="motionscript-engine.js"></script> <!-- 2. Add canvas --> <canvas id="myAnimation" style="width: 100%; height: 500px;"></canvas> <!-- 3. Initialize --> <script> const canvas = document.getElementById('myAnimation'); const engine = new MotionScriptEngine(canvas, { autoResize: true, antialias: true, onLoad: () => console.log('Animation loaded!'), onFrame: (data) => { // data.time, data.fps, data.progress } }); // Load your animation code const animationCode = ` scene { duration: 3s loop: true background: #000000 } object cube { type: cube position: 0,0,0 color: #00d9ff } animate cube { at 0s -> rotation: 0,0,0 at 3s -> rotation: 360,360,360 easing: linear } `; engine.loadScript(animationCode); engine.play(); </script>

🤖 AI Integration

Use with AI to generate animations from natural language:

async function generateAnimation(userPrompt) { // 1. Call AI with structured prompt const response = await fetch('/api/generate', { method: 'POST', body: JSON.stringify({ prompt: `Generate MotionScript code for: ${userPrompt} Format: scene { duration: 3s, loop: true, background: #000000 } object name { type: sphere, position: 0,0,0, color: #ff00ff } animate name { at 0s -> position: 0,0,0, at 3s -> position: 0,5,0, easing: linear }` }) }); const data = await response.json(); // 2. Load into engine engine.loadScript(data.code); engine.play(); } // Usage: generateAnimation("spinning neon cube")
